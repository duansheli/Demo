1. 通过docker创建mongo
#创建mongo持久化目录
mkdir -p /usr/local/graft/mongodata

docker pull mongo:3.4.6
docker run --name mongo -p 20107:27017 \
  -v /usr/local/graft/mongodata:/data/db \
  -d mongo:3.4.6 --auth --storageEngine wiredTiger

2. 进入容器
docker-enter mongo

3. 创建管理员
mongo 127.0.0.1:27017/admin
