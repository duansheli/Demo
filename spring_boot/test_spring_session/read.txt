1. 拉取镜像
docker pull redis:3.2.9

2. 设置redis参数
mkdir -p /data/redis/data_329
#添加密码
echo "requirepass pwd123" > /data/redis/redis.conf 

3. 启动容器
docker run -d  --name redis -p 6379:6379 \
-v /data/redis/data_325:/data \
-v /data/redis/redis.conf:/etc/redis.conf \
redis:3.2.9 redis-server /etc/redis.conf --appendonly yes
